# Install git
sudo dnf install -y git

# Install the necessary packages
sudo dnf install -y httpd php php-mysqli mariadb105

# Start and enable the Apache service
sudo systemctl enable httpd
sudo systemctl start httpd

# Add the ec2-user to the apache group
usermod -a -G apache ec2-user

# Set ownership and permissions for the web directory
sudo chown -R ec2-user:apache /var/www
sudo chmod 2775 /var/www

# Set permissions for directories and files within /var/www
sudo find /var/www -type d -exec chmod 2775 {} \;
sudo find /var/www -type f -exec chmod 0664 {} \;

# Clone or Copy files from your github repository
git clone https://github.com/george-848/aws-3_Tier_App.git
sudo cp -r home/ec2-user/aws-3_Tier_App/section8/resource-files/VeganStudioSourceCode/v5/*  /var/www/html
sudo rm home/ec2-user/aws-3_Tier_App --recursive
